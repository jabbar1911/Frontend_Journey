console.log("Day 88: Object Mastery");

// 1. Context & this
const user = {
    name: "Sairam",
    greet() {
        console.log(`Hi, I'm ${this.name}`);
    }
};
user.greet();

// 2. call, apply, bind
function intro(city) {
    console.log(`${this.name} in ${city}`);
}
intro.call(user, "Hyderabad");
intro.apply(user, ["New York"]);

// 3. Extraction
console.log("Keys:", Object.keys(user));

// 4. JS Quirks
console.log("false == []:", false == []);
