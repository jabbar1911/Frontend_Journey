console.log("Day 82: JavaScript Datatypes & Conversions");

// 1. Primitive Datatypes
let score = 100;
let userName = "Sairam";
let isLearning = true;
let nothing = null;
let notDefined;

console.log("Number check:", typeof score);
console.log("String check:", userName);
console.log("Boolean check:", isLearning);
console.log("Null check:", typeof nothing, "(Historical JS Bug)");
console.log("Undefined check:", typeof notDefined);

// 2. Type Conversion (Implicit & Explicit)
console.log("Type Conversion Lab:");
console.log("5 + '10' (Concatenation) =", 5 + "10");
console.log("'10' - 5 (Subtraction) =", "10" - 5);
console.log("Manual conversion:", Number("50"));

// 3. Reference vs Primitive
console.log("Reference Type Demonstration:");
let userArr = ["HTML", "CSS"];
let myArr = userArr; // Points to same memory
myArr.push("JS");
console.log("Original Array updated:", JSON.stringify(userArr));
