console.log("Day 87: Array Mastery");

// 1. Basic Operations
let fruits = ["Apple", "Banana", "Mango"];
console.log("Initial Array:", fruits);
console.log("Length:", fruits.length);

// 2. Add/Remove
fruits.push("Orange"); // Add end
fruits.unshift("Grape"); // Add start
fruits.splice(1, 1, "Watermelon"); // Replace
console.log("Modified Fruits:", fruits);

// 3. Sorting & Joining
let cities = ["Hyderabad", "New York", "London"];
console.log("Joined:", cities.sort().join(" | "));

// 4. Spread & Sets
let n1 = [1, 2, 2, 3];
let unique = [...new Set(n1)];
console.log("Unique List:", unique);

console.log("Array mastery module complete.");
